#Use the Return.annualized() function on your returns object to compute the annualized returns of each stock in your four stock portfolio. Which stock had the highest annualized return, and which the lowest?

# Load required libraries
library(PerformanceAnalytics)

# Calculate annualized returns
annualized_returns <- Return.annualized(returns)

# Print the annualized returns
print(annualized_returns)
